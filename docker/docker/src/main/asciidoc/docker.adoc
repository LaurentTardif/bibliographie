Now we can create a docker image. 
vi Dockerfile

[source]
----

FROM docker/whalesay:latest
RUN apt-get -y update && apt-get install -y fortunes
CMD /usr/games/fortune -a | cowsay
----
		  
docker build -t docker-whale .
